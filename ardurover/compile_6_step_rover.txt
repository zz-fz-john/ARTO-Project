#!/bin/bash
CURRENT_DIR=$(pwd)
ROOT_DIR="${CURRENT_DIR}/.."
echo "Current directory is: $CURRENT_DIR"
TOOLCHAIN_DIR="${ROOT_DIR}/toolchains/aarch32"
echo "Generating offline evidence"
cd ../pre-analysis16.0/util/generate_final_hash/ || { echo "Failed to enter generate_final_hash directory"; exit 1; }
echo "Generating configuration file"
python3 gen_config.py ../output/ardurover_output.S --binary-file ../output/ardurover_output --program-header ardurover_output.ph -o ../output/replay_single.cfg
echo "Re-executing database generation"
echo "-o: Target file"
echo "-c: Configuration file generated by gen_config.py"
echo "-bf: Binary file to be analyzed"
python3 generate_offline_database_no_threshold.py \
  -c ../output/replay_single.cfg \
  -l \
  -o ../output/hash_database_single_on_server_rover.txt \
  -bf ../output/ardurover_output \
  -dis ../output/ardurover_output.S \
  -IR ../output/after_insert_dummy_rover.ll \
  -svf ../../virtualcall/output/indirectcall_rover.txt \
  -tiff ../output/ToInsertFunc_rover.txt \
  -ocof ../output/only_called_once_func_backup_rover.txt \
  ../output/ardurover_output \
  -v -v -v
echo "Filtering out duplicate hash values"
python3 ReWriteDataBase.py -i ../output/hash_database_single_on_server.txt -o ../output/final_hash_database_single_on_server_rover.txt
