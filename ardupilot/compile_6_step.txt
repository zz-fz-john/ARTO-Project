#!/bin/bash
CURRENT_DIR=$(pwd)
ROOT_DIR="${CURRENT_DIR}/.."
echo "Current directory is: $CURRENT_DIR"
TOOLCHAIN_DIR="${ROOT_DIR}/toolchains/aarch32"
echo "Generating offline evidence"
cd ../pre-analysis16.0/util/generate_final_hash/ || { echo "Failed to enter simulated_executordirectory"; exit 1; }
echo "Generating configuration file"
python3 gen_config.py ../output/arducopter_output.S --binary-file ../output/arducopter_output --program-header arducopter_output.ph -o ../output/replay_single.cfg
echo "Re-executing database generation"
echo "-o: Target file"
echo "-c: Configuration file generated by gen_config.py"
echo "-bf: Binary file to be analyzed"


python3 generate_offline_database_no_threshold.py \
  -c ../output/replay_single.cfg \
  -l \
  -o ../output/hash_database_single_on_server.txt \
  -bf ../output/arducopter_output \
  -dis ../output/arducopter_output.S \
  -IR ../output/after_insert_dummy_copter.ll \
  -svf ../../virtualcall/output/indirectcall_copter.txt \
  -tiff ../output/ToInsertFunc_copter.txt \
  -ocof ../output/only_called_once_func_backup_copter.txt \
  ../output/arducopter_output \
  -v -v -v
echo "Filter out duplicate hash values"
python3 ReWriteDataBase.py -i ../output/hash_database_single_on_server.txt -o ../output/final_hash_database_single_on_server.txt
