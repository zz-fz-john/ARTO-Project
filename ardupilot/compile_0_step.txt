#!/bin/bash
echo  "set parameters"
CURRENT_DIR=$(pwd)
ROOT_DIR="${CURRENT_DIR}/.."
export CC="$HOME/llvm-arm-cross/llvm-project-16.0.0/build/bin/clang"
export CXX="$HOME/llvm-arm-cross/llvm-project-16.0.0/build/bin/clang++"
export CFLAGS="-O0 -g  -Xclang -no-opaque-pointers  --target=arm-linux-gnueabihf -gdwarf-4"
export CXXFLAGS="-O0 -g -Xclang -no-opaque-pointers  --target=arm-linux-gnueabihf -gdwarf-4"
export LDFLAGS="-fuse-ld=$HOME/llvm-arm-cross/llvm-project-16.0.0/build/bin/ld.lld"
LLVM_OPT="${ROOT_DIR}/llvm-16.0/llvm-project-16.0.0/build/bin/opt"
LLVM_DIS="${ROOT_DIR}/llvm-16.0/llvm-project-16.0.0/build/bin/llvm-dis"
LLVM_CHECKPOINT_PASS="${ROOT_DIR}/llvm-16.0/llvm-project-16.0.0/build/lib/LLVMInsertCheckPointsOnLoop.so"
LLVM_LIB="${ROOT_DIR}/llvm-16.0/llvm-project-16.0.0/build/lib"
TOOLCHAIN_DIR="${ROOT_DIR}/toolchains/aarch32"
echo  "run waf configure"
./waf configure --board=SITL_arm_linux_gnueabihf --disable-scripting
if [ $? -ne 0 ]; then
    echo "❌ waf configure failed! Please check the error messages."
    exit 1  
fi

echo "run waf copter"
./waf copter
if [ $? -ne 0 ]; then
    echo "❌ waf copter failed! Please check the error messages."
    exit 1
fi
echo "build completed successfully!"

echo "enter build directory"
cd build/SITL_arm_linux_gnueabihf/ || exit 1

echo "execute llvm-ar to process static libraries"
${ROOT_DIR}/llvm-16.0/llvm-project-16.0.0/build/bin/llvm-ar -s lib/libArduCopter_libs.a
if [ $? -ne 0 ]; then
    echo "❌ llvm-ar failed to process static libraries!"
    exit 1
fi
echo "executed llvm-ar"
${ROOT_DIR}/llvm-16.0/llvm-project-16.0.0/build/bin/clang++ -Wl,-plugin-opt=no-opaque-pointers  -Wl,--export-dynamic -fno-exceptions -Wl,--gc-sections -fuse-ld=lld -flto -Wl,--plugin-opt=emit-llvm -Wl,--lto-whole-program-visibility  ArduCopter/AP_Arming.cpp.*.o ArduCopter/AP_Rally.cpp.*.o ArduCopter/AP_State.cpp.*.o ArduCopter/Attitude.cpp.*.o ArduCopter/Copter.cpp.*.o ArduCopter/GCS_Copter.cpp.*.o ArduCopter/GCS_Mavlink.cpp.*.o ArduCopter/Log.cpp.*.o ArduCopter/Parameters.cpp.*.o ArduCopter/RC_Channel.cpp.*.o ArduCopter/UserCode.cpp.*.o ArduCopter/UserParameters.cpp.*.o ArduCopter/afs_copter.cpp.*.o ArduCopter/autoyaw.cpp.*.o ArduCopter/avoidance.cpp.*.o ArduCopter/avoidance_adsb.cpp.*.o ArduCopter/baro_ground_effect.cpp.*.o ArduCopter/commands.cpp.*.o ArduCopter/compassmot.cpp.*.o ArduCopter/crash_check.cpp.*.o ArduCopter/ekf_check.cpp.*.o ArduCopter/esc_calibration.cpp.*.o ArduCopter/events.cpp.*.o ArduCopter/failsafe.cpp.*.o ArduCopter/fence.cpp.*.o ArduCopter/heli.cpp.*.o ArduCopter/inertia.cpp.*.o ArduCopter/land_detector.cpp.*.o ArduCopter/landing_gear.cpp.*.o ArduCopter/mode.cpp.*.o ArduCopter/mode_acro.cpp.*.o ArduCopter/mode_acro_heli.cpp.*.o ArduCopter/mode_althold.cpp.*.o ArduCopter/mode_auto.cpp.*.o ArduCopter/mode_autorotate.cpp.*.o ArduCopter/mode_autotune.cpp.*.o ArduCopter/mode_avoid_adsb.cpp.*.o ArduCopter/mode_brake.cpp.*.o ArduCopter/mode_circle.cpp.*.o ArduCopter/mode_drift.cpp.*.o ArduCopter/mode_flip.cpp.*.o ArduCopter/mode_flowhold.cpp.*.o ArduCopter/mode_follow.cpp.*.o ArduCopter/mode_guided.cpp.*.o ArduCopter/mode_guided_nogps.cpp.*.o ArduCopter/mode_land.cpp.*.o ArduCopter/mode_loiter.cpp.*.o ArduCopter/mode_poshold.cpp.*.o ArduCopter/mode_rtl.cpp.*.o ArduCopter/mode_smart_rtl.cpp.*.o ArduCopter/mode_sport.cpp.*.o ArduCopter/mode_stabilize.cpp.*.o ArduCopter/mode_stabilize_heli.cpp.*.o ArduCopter/mode_systemid.cpp.*.o ArduCopter/mode_throw.cpp.*.o ArduCopter/mode_turtle.cpp.*.o ArduCopter/mode_zigzag.cpp.*.o ArduCopter/motor_test.cpp.*.o ArduCopter/motors.cpp.*.o ArduCopter/navigation.cpp.*.o ArduCopter/precision_landing.cpp.*.o ArduCopter/radio.cpp.*.o ArduCopter/sensors.cpp.*.o ArduCopter/standby.cpp.*.o ArduCopter/surface_tracking.cpp.*.o ArduCopter/system.cpp.*.o ArduCopter/takeoff.cpp.*.o ArduCopter/takeoff_check.cpp.*.o ArduCopter/terrain.cpp.*.o ArduCopter/toy_mode.cpp.*.o ArduCopter/tuning.cpp.*.o -Llib -lArduCopter_libs -o arducopter.bc
$LLVM_DIS -opaque-pointers=0 arducopter.bc   -o arducopter.ll
echo "successfully linked to bc file"
echo "---------------------------------"
echo "processing IR file"

echo "optimize pass to avoid redundant indirect calls and remove function aliases"
$LLVM_OPT -S  -opaque-pointers=0 -passes="globalopt" -o arducopter_modify.ll arducopter.ll

echo "use whole-program devirtualization pass to remove virtual function calls with only one target" 
$LLVM_OPT -S -passes=wholeprogramdevirt -whole-program-visibility -wholeprogramdevirt-summary-action=export -opaque-pointers=0 arducopter_modify.ll -o after_devirt.ll
 
echo "merge multiple returns in functions to align with SVF statements" 
$LLVM_OPT -S  -passes="mergereturn" -opaque-pointers=0 after_devirt.ll -o after_merge.ll


echo "expand nested GEP structures in IR to align with SVF statements and identify key operations"
$LLVM_OPT -load $LLVM_CHECKPOINT_PASS --Break-Constant-GEPs -enable-new-pm=0 -opaque-pointers=0 after_merge.ll -o arducopter_breakconstant.bc

$LLVM_DIS -opaque-pointers=0 arducopter_breakconstant.bc -o arducopter_breakconstant.ll

echo "generate original flight controller for comparison, with checkpoints inserted to record timing"
$LLVM_OPT -load $LLVM_CHECKPOINT_PASS --get-instrument-critical-func -enable-new-pm=0 -opaque-pointers=0 arducopter_breakconstant.bc -o enable_start_collect.bc

$LLVM_DIS -opaque-pointers=0 enable_start_collect.bc -o enable_start_collect.ll

# generate original flight controller
$HOME/llvm-arm-cross/llvm-project-16.0.0/build/bin/llc -O3 -filetype=obj enable_start_collect.bc -mtriple=arm-linux-gnueabihf -mcpu=cortex-a53  -o enable_start_collect.o || { echo "llc failed"; exit 1; }

../../../toolchains/aarch32/arm-linux-gnueabihf/bin/ld -EL -z relro -X --hash-style=gnu --eh-frame-hdr -m armelf_linux_eabi -dynamic-linker /lib/ld-linux-armhf.so.3 -o arducopter_org $TOOLCHAIN_DIR/arm-linux-gnueabihf/libc/usr/lib/Scrt1.o $TOOLCHAIN_DIR/arm-linux-gnueabihf/libc/usr/lib/crti.o $TOOLCHAIN_DIR/lib/gcc/arm-linux-gnueabihf/8.2.1/crtbeginS.o -L/usr/lib/gcc-cross/arm-linux-gnueabihf/9 -L/usr/lib/gcc-cross/arm-linux-gnueabihf/9/../../../../arm-linux-gnueabihf/lib/../lib -L/usr/lib/gcc-cross/arm-linux-gnueabihf/9/../../../../lib -L/lib/arm-linux-gnueabihf -L/lib/../lib -L/usr/lib/arm-linux-gnueabihf -L/usr/lib/../lib -L/usr/lib/gcc-cross/arm-linux-gnueabihf/9/../../../../arm-linux-gnueabihf/lib -L/lib -L/usr/lib -plugin $LLVM_LIB/LLVMgold.so -plugin-opt=mcpu=arm1176jzf-s --gc-sections -Bstatic  enable_start_collect.o  ../../../measurement-engine/first-measure-in-secure-world/output/trampoline.o ../../../measurement-engine/first-measure-in-secure-world/output/CFeventSingleThread.o ../../../measurement-engine/first-measure-in-secure-world/output/data_flow.o ../../../measurement-engine/first-measure-in-secure-world/output/heap_section.o  ../../../measurement-engine/first-measure-in-secure-world/output/xxhash3.o   ../../../measurement-engine/first-measure-in-secure-world/output/dummycode.o -Bdynamic $TOOLCHAIN_DIR/arm-linux-gnueabihf/libc/lib/libm.so.6 $TOOLCHAIN_DIR/arm-linux-gnueabihf/libc/lib/libdl.so.2 $TOOLCHAIN_DIR/arm-linux-gnueabihf/lib/libstdc++.so.6 $TOOLCHAIN_DIR/arm-linux-gnueabihf/libc/lib/libm.so.6 $TOOLCHAIN_DIR/arm-linux-gnueabihf/lib/libgcc_s.so.1  $TOOLCHAIN_DIR/lib/gcc/arm-linux-gnueabihf/8.2.1/libgcc.a -lpthread -lc $TOOLCHAIN_DIR/arm-linux-gnueabihf/lib/libgcc_s.so.1 $TOOLCHAIN_DIR/lib/gcc/arm-linux-gnueabihf/8.2.1/libgcc.a /usr/lib/gcc-cross/arm-linux-gnueabihf/9/crtendS.o $TOOLCHAIN_DIR/arm-linux-gnueabihf/libc/usr/lib/crtn.o 

echo "generate avoid_handle_function.txt and to_process_function.txt"
$LLVM_OPT -load $LLVM_CHECKPOINT_PASS  --generate-process-function -enable-new-pm=0 -opaque-pointers=0 arducopter_breakconstant.ll

echo "completed IR preprocessing"

echo "✅ build completed successfully!"
cd ../../
